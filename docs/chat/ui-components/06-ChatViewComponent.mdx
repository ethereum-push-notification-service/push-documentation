- **ChatViewComponent**
    
    ### want to add the entire chat component in your website but donâ€™t have the time to code the entire message body UI ? import ChatViewComponent, pass the chatId and there you have it !
    
    - Inputs
        
        ```
          messageInput?: boolean; // default true
          chatViewList?: boolean; // default true
          chatProfile?: boolean; // default true
          chatId: string;
          limit?: number;
          emoji?: boolean; // default true
          gif?: boolean; // default true
          file?: boolean;  // default true
          isConnected?: boolean;  // default true
          onClick?: (() => void()) // use this to pass a function when the user is not verified as per the rules of access control
        ```

          );
        };
        
        export default ChatViewComponentTest;
